62 channel_ref_t* broadcast_join(broadcast_t* broadcast, channel_ref_t* channel_ref) {
63  dlist_node_t* node = 0;
64  /* 创建新的引用 */
65  channel_ref_t* channel_shared = channel_ref_share(channel_ref);
66  assert(channel_shared);
67  lock_lock(broadcast->lock);
68  /* 添加到广播域链表，设置链表节点（快速删除） */
69  node = dlist_add_tail_node(broadcast->channels, channel_shared);
70  lock_unlock(broadcast->lock);
71  channel_ref_set_domain_node(channel_shared, node);
72  channel_ref_set_domain_id(channel_shared, broadcast->domain_id);
73  return channel_shared;
74 }

76 int broadcast_leave(broadcast_t* broadcast, channel_ref_t* channel_ref) {
77  dlist_node_t* node = 0;
78  assert(broadcast);
79  assert(channel_ref);
80  assert(channel_ref_check_share(channel_ref));
81  if (broadcast->domain_id != channel_ref_get_domain_id(channel_ref)) {
82  return error_not_correct_domain;
83  }
84  node = channel_ref_get_domain_node(channel_ref);
85  assert(node);
86  lock_lock(broadcast->lock);
87  dlist_delete(broadcast->channels, node);
88  lock_unlock(broadcast->lock);
89  /* 销毁引用 */
90  channel_ref_leave(channel_ref);
91  return error_ok;
92 }

104 int broadcast_write(broadcast_t* broadcast, char* buffer, uint32_t size) {
105  dlist_node_t* node = 0;
106  dlist_node_t* temp = 0;
107  channel_ref_t* channel_ref = 0;
108  int error = 0;
109  int count = 0;
110  assert(broadcast);
111  assert(broadcast->channels);
112  lock_lock(broadcast->lock);
113  dlist_for_each_safe(broadcast->channels, node, temp) {
114  channel_ref = (channel_ref_t*)dlist_node_get_data(node);
115  error = channel_ref_write(channel_ref, buffer, size);
116  if (error != error_ok) {
117  /* 销毁发送失败的管道 */
118  broadcast_leave(broadcast, channel_ref);
119  } else {
120  count++;
121  }
122  }
123  lock_unlock(broadcast->lock);
124  return count;
125 }
