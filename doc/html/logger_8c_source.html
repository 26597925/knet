38 logger_t* logger_create(const char* path, logger_level_e level, logger_mode_e mode) {
39  char temp[PATH_MAX] = {0};
40  logger_t* logger = create(logger_t);
41  memset(logger, 0, sizeof(logger_t));
42  logger->mode = mode;
43  logger->level = level;
44  logger->lock = lock_create();
45  assert(logger->lock);
46  if (!path) {
47  /* 日志建立在当前目录 */
48  path = path_getcwd(temp, sizeof(temp));
49  strcat(temp, "/knet.log");
50  }
51  if (mode & logger_mode_file) {
52  assert(path);
53  if (mode & logger_mode_override) {
54  /* 打开并清空 */
55  logger->fd = fopen(path, "w+");
56  } else {
57  /* 添加到原有日志 */
58  logger->fd = fopen(path, "a+");
59  }
60  if (!logger->fd) {
61  goto fail_return;
62  }
63  }
64  return logger;
65 fail_return:
66  destroy(logger);
67  return 0;
68 }

79 int logger_write(logger_t* logger, logger_level_e level, const char* format, ...) {
80  char buffer[64] = {0};
81  int bytes = 0;
82  static const char* logger_level_name[] = { 0, "VERB", "INFO", "WARN", "ERRO", "FATA" };
83  va_list arg_ptr;
84  assert(logger);
85  assert(format);
86  if (logger->level > level) {
87  /* 日志等级不足 */
88  return error_ok;
89  }
90  va_start(arg_ptr, format);
91  time_get_string(buffer, sizeof(buffer));
92  if (logger->mode & logger_mode_file) {
93  /* 写入日志文件 */
94  lock_lock(logger->lock);
95  fprintf(logger->fd, "[%s][%s]", logger_level_name[level], buffer);
96  bytes = vfprintf(logger->fd, format, arg_ptr);
97  if (bytes <= 0) {
98  lock_unlock(logger->lock);
99  return error_logger_write;
100  }
101  fprintf(logger->fd, "\n");
102  if (logger->mode & logger_mode_flush) {
103  /* 立即写入 */
104  fflush(logger->fd);
105  }
106  lock_unlock(logger->lock);
107  }
108  if (logger->mode & logger_mode_console) {
109  /* 写入stderr */
110  lock_lock(logger->lock);
111  fprintf(stderr, "[%s][%s]", logger_level_name[level], buffer);
112  vfprintf(stderr, format, arg_ptr);
113  fprintf(stderr, "\n");
114  if (logger->mode & logger_mode_flush) {
115  /* 立即写入 */
116  fflush(stderr);
117  }
118  lock_unlock(logger->lock);
119  }
120  va_end(arg_ptr);
121  return error_ok;
122 }
