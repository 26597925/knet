137 hash_t* hash_create(uint32_t size, hash_dtor_t dtor) {
138  uint32_t i = 0;
139  hash_t* hash = create(hash_t);
140  assert(hash);
141  memset(hash, 0, sizeof(hash_t));
142  if (!size) {
143  size = 16; /* 默认bucket数量 */
144  }
145  hash->dtor = dtor;
146  hash->size = size;
147  hash->buckets = (dlist_t**)create_type(dlist_t*, sizeof(dlist_t*) * size);
148  assert(hash->buckets);
149  memset(hash->buckets, 0, size * sizeof(dlist_t*));
150  /* 创建bucket链表 */
151  for (; i < size; i++) {
152  hash->buckets[i] = dlist_create();
153  }
154  return hash;
155 }

157 void hash_destroy(hash_t* hash) {
158  uint32_t i = 0;
159  dlist_node_t* node = 0;
160  dlist_node_t* temp = 0;
161  hash_value_t* value = 0;
162  assert(hash);
163  /* 遍历所有bucket并销毁 */
164  for (; i < hash->size; i++) {
165  if (hash->buckets[i]) {
166  dlist_for_each_safe(hash->buckets[i], node, temp) {
167  value = (hash_value_t*)dlist_node_get_data(node);
168  if (hash->dtor) {
169  /* 自定义销毁 */
170  hash->dtor(value->value);
171  }
172  /* 销毁值 */
173  hash_value_destroy(value);
174  }
175  /* 销毁链表 */
176  dlist_destroy(hash->buckets[i]);
177  hash->buckets[i] = 0;
178  }
179  }
180  /* 销毁bucket */
181  destroy(hash->buckets);
182  destroy(hash);
183 }

213 void* hash_remove(hash_t* hash, uint32_t key) {
214  uint32_t hash_key = 0;
215  dlist_node_t* node = 0;
216  dlist_node_t* temp = 0;
217  hash_value_t* hash_value = 0;
218  void* value = 0;
219  assert(hash);
220  hash_key = key % hash->size;
221  /* 遍历链表查找 */
222  dlist_for_each_safe(hash->buckets[hash_key], node, temp) {
223  if (node == hash->it_node_next) { /* 删除正在遍历节点的后续节点 */
224  hash->it_node_next = dlist_next(hash->buckets[hash_key], hash->it_node_next);
225  }
226  hash_value = (hash_value_t*)dlist_node_get_data(node);
227  if (hash_value_equal(hash_value, key)) {
228  hash->count--;
229  value = hash_value_get_value(hash_value);
230  /* 销毁节点 */
231  dlist_delete(hash->buckets[hash_key], node);
232  /* 销毁值 */
233  hash_value_destroy(hash_value);
234  return value; /* 返回给调用者 */
235  }
236  }
237  return 0; /* 没找到 */
238 }

337 hash_value_t* hash_get_first(hash_t* hash) {
338  uint32_t i = 0;
339  dlist_t* list = 0;
340  hash_value_t* hash_value = 0;
341  assert(hash);
342  hash->it_index = 0;
343  hash->it_node_safe = 0;
344  hash->it_node_next = 0;
345  if (!hash->count) { /* 没有元素 */
346  return 0;
347  }
348  /* 找到第一个有元素的桶 */
349  for (; i < hash->size; i++) {
350  list = hash->buckets[i];
351  assert(list);
352  if (!dlist_empty(list)) {
353  hash->it_index = i;
354  break;
355  }
356  }
357  assert(list);
358  hash->it_node_safe = dlist_get_front(list);
359  assert(hash->it_node_safe);
360  /* 预先取下一个元素，不论是否有元素 */
361  hash->it_node_next = dlist_next(list, hash->it_node_safe);
362  hash_value = (hash_value_t*)dlist_node_get_data(hash->it_node_safe);
363  assert(hash_value);
364  return hash_value;
365 }

367 hash_value_t* hash_next(hash_t* hash) {
368  uint32_t i = 0;
369  dlist_t* list = 0;
370  hash_value_t* hash_value = 0;
371  assert(hash);
372  if (!hash->it_node_safe) { /* 未调用hash_get_first() */
373  assert(0);
374  return 0;
375  }
376  if (hash->it_node_next) { /* 当前桶还有下一个元素 */
377  hash->it_node_safe = hash->it_node_next; /* 交换 */
378  } else { /* 到下一个有元素的桶 */
379  if (hash->it_index >= hash->size) {
380  return 0;
381  }
382  /* 未到达尾部 */
383  hash->it_index++;
384  /* 遍历后续桶 */
385  for (i = hash->it_index; i < hash->size; i++) {
386  list = hash->buckets[i];
387  assert(list);
388  if (!dlist_empty(list)) {
389  hash->it_index = i;
390  break;
391  }
392  }
393  if (!list) { /* 后续桶都为空 */
394  return 0;
395  }
396  /* 取当前元素 */
397  hash->it_node_safe = dlist_get_front(list);
398  assert(hash->it_node_safe);
399  }
400  /* 预先取下一个元素，不论是否有元素 */
401  hash->it_node_next = dlist_next(list, hash->it_node_safe);
402  hash_value = (hash_value_t*)dlist_node_get_data(hash->it_node_safe);
403  assert(hash_value);
404  return hash_value;
