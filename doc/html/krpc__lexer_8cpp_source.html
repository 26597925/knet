84 krpc_token_t* krpc_lexer_t::get_token() {
85  static const int MAX_TOKEN_SIZE = 64;
86  char token[MAX_TOKEN_SIZE] = {0};
87  int i = 0;
88  int type = 0;
89  char c = 0;
90  memset(token, 0, sizeof(token));
91  if (!verify()) {
92  return 0;
93  }
94  eat_whites();
95  for (c = current(); (c) && (!check_terminator(c, _icomment)); c = forward(1), i++) {
96  token[i] = c;
97  }
98  if (i >= MAX_TOKEN_SIZE) {
99  raise_exception("reach max name length");
100  }
101  if (!token[0] && check_terminator(c, false)) {
102  token[i++] = current();
103  }
104  if (!token[0]) {
105  return 0;
106  }
107  switch (token[0]) {
108  case 'o':
109  if (check_keyword6(token, 'o', 'b', 'j', 'e', 'c', 't')) {
110  type = krpc_token_object;
111  add_col(6);
112  }
113  break;
114  case 'i':
115  if (check_keyword2(token, 'i', '8')) {
116  type = krpc_token_i8;
117  add_col(2);
118  } else if (check_keyword3(token, 'i', '1', '6')) {
119  type = krpc_token_i16;
120  add_col(3);
121  } else if (check_keyword3(token, 'i', '3', '2')) {
122  type = krpc_token_i32;
123  add_col(3);
124  } else if (check_keyword3(token, 'i', '6', '4')) {
125  type = krpc_token_i64;
126  add_col(3);
127  } else if (check_keyword6(token, 'i', 'm', 'p', 'o', 'r', 't')) {
128  type = krpc_token_import;
129  add_col(6);
130  }
131  break;
132  case 'u':
133  if (check_keyword3(token, 'u', 'i', '8')) {
134  type = krpc_token_ui8;
135  add_col(3);
136  } else if (check_keyword4(token, 'u', 'i', '1', '6')) {
137  type = krpc_token_ui16;
138  add_col(4);
139  } else if (check_keyword4(token, 'u', 'i', '3', '2')) {
140  type = krpc_token_ui32;
141  add_col(4);
142  } else if (check_keyword4(token, 'u', 'i', '6', '4')) {
143  type = krpc_token_ui64;
144  add_col(4);
145  }
146  break;
147  case 'f':
148  if (check_keyword3(token, 'f', '3', '2')) {
149  type = krpc_token_f32;
150  add_col(3);
151  } else if (check_keyword3(token, 'f', '6', '4')) {
152  type = krpc_token_f64;
153  add_col(3);
154  }
155  break;
156  case 's':
157  if (check_keyword6(token, 's', 't', 'r', 'i', 'n', 'g')) {
158  type = krpc_token_string;
159  add_col(6);
160  }
161  break;
162  case '{':
163  forward(1);
164  add_col(1);
165  type = krpc_token_left_brace;
166  break;
167  case '}':
168  forward(1);
169  add_col(1);
170  type = krpc_token_right_brace;
171  break;
172  case '(':
173  forward(1);
174  add_col(1);
175  type = krpc_token_left_round;
176  break;
177  case ')':
178  forward(1);
179  add_col(1);
180  type = krpc_token_right_round;
181  break;
182  case 'r':
183  if (check_keyword3(token, 'r', 'p', 'c')) {
184  type = krpc_token_rpc;
185  add_col(3);
186  }
187  break;
188  case ',':
189  forward(1);
190  add_col(1);
191  type = krpc_token_comma;
192  break;
193  case '[':
194  if (has_next() && (_stream[1] == ']')) {
195  // 数组
196  forward(2);
197  add_col(2);
198  type = krpc_token_array;
199  } else if (has_next() && (_stream[1] == '#')) {
200  // 嵌入注释
201  _icomment = true;
202  forward(2);
203  add_col(2);
204  type = krpc_token_inline_comment;
205  } else {
206  raise_exception("invalid array declaration");
207  }
208  break;
209  case ']':
210  _icomment = false;
211  forward(1);
212  add_col(1);
213  type = krpc_token_right_square;
214  break;
215  case '/':
216  // 单行/多行注释
217  forward(1);
218  eat_comment();
219  return get_token();
220  default:
221  break;
222  }
223  if (0 == type) {
224  // 变量名，对象名
225  add_col(i);
226  if (!_icomment) {
227  if (!check_var_name(token)) {
228  raise_exception("variable name must start by '_' or letter");
229  }
230  }
231  type = krpc_token_text;
232  }
233  return new krpc_token_t(token, _row, _col, type);
234 }

272 void krpc_lexer_t::eat_whites() {
273  char c = 0;
274  for (c = current(); (c); c = forward(1)) {
275  switch (c) {
276  case ' ':
277  case '\r':
278  add_col(1);
279  break;
280  case '\t':
281  add_col(4);
282  break;
283  case '\n':
284  _col = 1;
285  inc_row();
286  break;
287  default:
288  return;
289  }
290  }
291 }

301 bool krpc_lexer_t::check_terminator(char c, bool icomment) {
302  if (icomment) {
303  if (c != ']') {
304  return false;
305  }
306  } else {
307  switch (c) {
308  case '{': case '}': case '(': case ')': case ',': case '[': case ']': case '/':
309  return true;
310  }

312  }

340 void krpc_lexer_t::eat_comment() {
341  check_raise_exception(verify(), "uncomplete comment");
342  if (current() == '/') {
343  // 单行注释
344  for (; verify(); forward(1), add_col(1)) {
345  if (current() == '\n') {
346  _col = 1;
347  inc_row();
348  forward(1);
349  return;
350  }
351  }
352  } else {
353  raise_exception("invalid comment");
354  }
355  eat_whites();
356 }
