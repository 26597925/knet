56 int _select(loop_t* loop) {
57  socket_t max_fd = 0;
58  socket_t fd = 0;
59  int error = 0;
60  dlist_node_t* node = 0;
61  dlist_node_t* temp = 0;
62  channel_ref_t* channel_ref = 0;
63  struct timeval tv = {0, 100}; /* 空转时最多等待1ms */
64  loop_select_t* impl = (loop_select_t*)loop_get_impl(loop);
65  FD_ZERO(impl->read_fds);
66  FD_ZERO(impl->send_fds);
67  dlist_for_each_safe(loop_get_active_list(loop), node, temp) {
68  channel_ref = (channel_ref_t*)dlist_node_get_data(node);
69  fd = channel_ref_get_socket_fd(channel_ref);
70  if (channel_ref_check_balance(channel_ref)) {
71  }
72  if (channel_ref_check_event(channel_ref, channel_event_recv)) {
73  FD_SET(fd, impl->read_fds);
74  max_fd = max(max_fd, fd);
75  }
76  if (channel_ref_check_event(channel_ref, channel_event_send)) {
77  FD_SET(fd, impl->send_fds);
78  max_fd = max(max_fd, fd);
79  }
80  }
81  error = select(max_fd + 1, impl->read_fds, impl->send_fds, 0, &tv);
82  if (0 > error) {
83  return error_loop_fail;
84  }
85  return error_ok;
86 }

88 int impl_run_once(loop_t* loop) {
89  dlist_node_t* node = 0;
90  dlist_node_t* temp = 0;
91  channel_ref_t* channel_ref = 0;
92  socket_t fd = 0;
93  time_t ts = time(0);
94  loop_select_t* impl = (loop_select_t*)loop_get_impl(loop);
95  int error = _select(loop);
96  if (error != error_ok) {
97  return error;
98  }
99  dlist_for_each_safe(loop_get_active_list(loop), node, temp) {
100  channel_ref = (channel_ref_t*)dlist_node_get_data(node);
101  fd = channel_ref_get_socket_fd(channel_ref);
102  if (FD_ISSET(fd, impl->read_fds)) {
103  channel_ref_update(channel_ref, channel_event_recv, ts);
104  }
105  if (FD_ISSET(fd, impl->send_fds)) {
106  channel_ref_update(channel_ref, channel_event_send, ts);
107  }
108  }
109  loop_check_timeout(loop, ts);
110  loop_check_close(loop);
111  return error_ok;
112 }
