99 int krpc_proc(krpc_t* rpc, stream_t* stream) {
100  int available = 0; /* 管道内可读字节数 */
101  uint16_t length = 0; /* unmarshal字节数*/
102  krpc_object_t* o = 0; /* unmarshal得到的对象 */
103  krpc_cb_t cb = 0; /* 回调函数 */
104  int error_cb = 0; /* 回调函数返回值 */
105  int error = error_ok; /* 本函数返回值 */
106  krpc_header_t header; /* RPC协议头 */
107  assert(rpc);
108  assert(stream);
109  memset(&header, 0, sizeof(krpc_header_t));
110  available = stream_available(stream);
111  if (available < sizeof(krpc_header_t)) {
112  /* 字节数不够 */
113  return error_rpc_not_enough_bytes;
114  }
115  if (error_ok != stream_copy(stream, &header, sizeof(header))) {
116  return error_rpc_unmarshal_fail;
117  }
118  if (header.length > available) {
119  /* 字节数不够 */
120  return error_rpc_not_enough_bytes;
121  } else {
122  if (error_ok != stream_eat(stream, sizeof(header))) {
123  return error_rpc_unmarshal_fail;
124  }
125  }
126  /* unmarshal */
127  error = krpc_object_unmarshal(stream, &o, &length);
128  if (error_ok != error) {
129  error = error_rpc_unmarshal_fail;
130  goto error_return;
131  }
132  /* 调用回调 */
133  if ((header.type == krpc_call_type_call) || (header.type == krpc_call_type_result)) {
134  /* 调用/返回 */
135  cb = krpc_get_cb(rpc, header.rpcid);
136  if (cb == 0) {
137  error = error_rpc_unknown_id;
138  goto error_return;
139  }
140  error_cb = cb(o);
141  if (error_cb == rpc_error) {
142  /* 错误 */
143  error = error_rpc_cb_fail;
144  goto error_return;
145  } else if (error_cb == rpc_error_close) {
146  /* 错误并关闭 */
147  error = error_rpc_cb_fail_close;
148  goto error_return;
149  } else if (error_cb == rpc_close) {
150  /* 要求关闭 */
151  error = error_rpc_cb_close;
152  goto error_return;
153  } else if (error_cb == rpc_ok) {
154  /* ok */
155  goto error_return;
156  } else {
157  /* 未知返回值，忽略 */
158  }
159  } else {
160  /* 错误类型 */
161  error = error_rpc_unknown_type;
162  }
163 error_return:
164  if (o) {
165  krpc_object_destroy(o);
166  }
167  return error;
168 }

170 int krpc_call(krpc_t* rpc, stream_t* stream, uint16_t rpcid, krpc_object_t* o) {
171  krpc_header_t header; /* RPC协议头 */
172  assert(rpc);
173  assert(stream);
174  assert(rpcid);
175  assert(o);
176  memset(&header, 0, sizeof(krpc_header_t));
177  header.rpcid = rpcid;
178  header.type = krpc_call_type_call;
179  header.length = sizeof(krpc_header_t) + krpc_object_get_marshal_size(o);
180  /* 发送协议头 */
181  if (error_ok != stream_push(stream, &header, sizeof(header))) {
182  return error_rpc_marshal_fail;
183  }
184  /* marshal */
185  return krpc_object_marshal(o, stream, 0);
186 }
