99 AcceptEx_t* socket_data_prepare_accept(per_sock_t* data) {
100  socket_t fd = 0;
101  assert(data);
102  fd = channel_ref_get_socket_fd(data->channel_ref);
103  if (!data->AcceptEx_info) {
104  data->AcceptEx_info = create(AcceptEx_t);
105  assert(data->AcceptEx_info);
106  memset(data->AcceptEx_info, 0, sizeof(AcceptEx_t));
107  }
108  if (!data->AcceptEx_info->fn_AcceptEx) {
109  data->AcceptEx_info->fn_AcceptEx = get_fn_AcceptEx(fd);
110  }
111  /* 建立一个支持重叠I/O的套接字 */
112  data->AcceptEx_info->socket_fd = WSASocket(AF_INET, SOCK_STREAM, IPPROTO_TCP, 0, 0, WSA_FLAG_OVERLAPPED);
113  if (data->AcceptEx_info->socket_fd == INVALID_SOCKET) {
114  assert(0);
115  return 0;
116  }
117  return data->AcceptEx_info;
118 }

151 int _select(loop_t* loop, time_t ts) {
152  BOOL error = FALSE;
153  DWORD bytes = 0;
154  DWORD last_error = 0;
155  per_io_t* per_io = 0;
156  per_sock_t* per_sock = 0;
157  channel_ref_t* channel_ref = 0;
158  loop_iocp_t* impl = get_impl(loop);
159  error = GetQueuedCompletionStatus(impl->iocp, &bytes, (PULONG_PTR)&per_sock, (LPOVERLAPPED*)&per_io, 1);
160  last_error = GetLastError();
161  if (FALSE == error) {
162  if ((last_error == WAIT_TIMEOUT) || (last_error == ERROR_OPERATION_ABORTED)) {
163  return error_ok;
164  }
165  }
166  assert(per_sock);
167  assert(per_io);
168  channel_ref = per_sock->channel_ref;
169  assert(channel_ref);
170  if ((per_io->type & io_type_recv) || (per_io->type & io_type_accept)) {
171  channel_ref_update(channel_ref, channel_event_recv, ts);
172  } else if ((per_io->type & io_type_send) || (per_io->type & io_type_connect)) {
173  channel_ref_update(channel_ref, channel_event_send, ts);
174  }
175  return error_ok;
176 }

191 socket_t impl_channel_accept(channel_ref_t* channel_ref) {
192  socket_t acceptor = 0;
193  per_sock_t* per_sock = 0;
194  socket_t client = 0;
195  assert(channel_ref);
196  acceptor = channel_ref_get_socket_fd(channel_ref);
197  assert(acceptor);
198  per_sock = (per_sock_t*)channel_ref_get_data(channel_ref);
199  assert(per_sock);
200  assert(per_sock->AcceptEx_info);
201  client = per_sock->AcceptEx_info->socket_fd;
202  /* MSDN: On Windows XP and later, once the AcceptEx function completes and the SO_UPDATE_ACCEPT_CONTEXT
203  option is set on the accepted socket, the local address associated with the accepted socket can also
204  be retrieved using the getsockname function. Likewise, the remote address associated with the accepted
205  socket can be retrieved using the getpeername function.
206  */
207  setsockopt(client, SOL_SOCKET, SO_UPDATE_ACCEPT_CONTEXT, (char*)&acceptor, sizeof(socket_t));
208  return client;
209 }

226 void on_iocp_recv(channel_ref_t* channel_ref) {
227  DWORD bytes = 0;
228  DWORD flags = 0;
229  DWORD error = 0;
230  WSABUF sbuf = {0, 0};
231  BOOL result = FALSE;
232  socket_t fd = channel_ref_get_socket_fd(channel_ref);
233  int flag = channel_ref_get_flag(channel_ref);
234  per_sock_t* per_sock = (per_sock_t*)channel_ref_get_data(channel_ref);
235  AcceptEx_t* AcceptEx_ptr = 0;
236  per_io_t* per_io = &per_sock->io_recv;
237  if (channel_ref_check_state(channel_ref, channel_state_accept)) {
238  per_io->type = io_type_accept;
239  AcceptEx_ptr = socket_data_prepare_accept(per_sock);
240  /* 投递一个accept请求 */
241  result = AcceptEx_ptr->fn_AcceptEx(fd, AcceptEx_ptr->socket_fd, AcceptEx_ptr->buffer,
242  0, ACCEPTEX_ADDR_SIZE, ACCEPTEX_ADDR_SIZE, 0, &per_io->ov);
243  if (result == FALSE) {
244  error = GetLastError();
245  if (error != ERROR_IO_PENDING) {
246  channel_ref_close(channel_ref);
247  }
248  return;
249  }
250  } else {
251  per_io->type = io_type_recv;
252  /* 投递一个0长度recv请求 */
253  result = WSARecv(fd, &sbuf, 1, &bytes, &flags, &per_io->ov, 0);
254  if (result != 0) {
255  error = GetLastError();
256  if ((error != ERROR_IO_PENDING) && (error != WSAENOTCONN)) {
257  channel_ref_close(channel_ref);
258  }
259  return;
260  }
261  }
262  flag |= io_type_recv;
263  channel_ref_set_flag(channel_ref, flag);
264 }

266 void on_iocp_send(channel_ref_t* channel_ref) {
267  DWORD bytes = 0;
268  DWORD flags = 0;
269  DWORD error = 0;
270  WSABUF sbuf = {0, 0};
271  int result = FALSE;
272  socket_t fd = channel_ref_get_socket_fd(channel_ref);
273  int flag = channel_ref_get_flag(channel_ref);
274  per_sock_t* per_sock = (per_sock_t*)channel_ref_get_data(channel_ref);
275  per_io_t* per_io = &per_sock->io_send;
276  /* 设置投递事件类型 */
277  if (channel_ref_check_state(channel_ref, channel_state_connect)) {
278  per_io->type = io_type_connect;
279  } else {
280  per_io->type = io_type_send;
281  }
282  /* 连接状态的管道，检查是否已经完成 */
283  if (channel_ref_check_state(channel_ref, channel_state_connect)) {
284  if (!socket_check_send_ready(channel_ref_get_socket_fd(channel_ref))) {
285  /* 未完成连接不投递send请求 */
286  return;
287  }
288  }
289  /* 投递一个0长度send请求 */
290  result = WSASend(fd, &sbuf, 1, &bytes, flags, &per_io->ov, 0);
291  if (result != 0) {
292  error = GetLastError();
293  if ((error != ERROR_IO_PENDING) && (error != WSAENOTCONN)) {
294  channel_ref_close(channel_ref);
295  }
296  return;
297  }
298  /* 设置投递标志 */
299  flag |= io_type_send;
300  channel_ref_set_flag(channel_ref, flag);
301 }

303 int impl_event_add(channel_ref_t* channel_ref, channel_event_e e) {
304  assert(channel_ref);
305  if (channel_ref_check_state(channel_ref, channel_state_close)) {
306  return error_already_close;
307  }
308  /* 投递事件 */
309  if (channel_event_recv & e) {
310  on_iocp_recv(channel_ref);
311  }
312  if (e & channel_event_send) {
313  on_iocp_send(channel_ref);
314  }
315  return error_ok;

318 int impl_event_remove(channel_ref_t* channel_ref, channel_event_e e) {
319  int flag = 0;
320  per_sock_t* per_sock = 0;
321  e;
322  assert(channel_ref);
323  flag = channel_ref_get_flag(channel_ref);
324  if (flag & io_type_recv) {
325  flag &= ~io_type_recv;
326  }
327  if (flag & io_type_send) {
328  flag &= ~io_type_send;
329  }
330  /* 同时取消读写即关闭 */
331  if ((e & channel_event_recv) && (e & channel_event_send)) {
332  per_sock = get_data(channel_ref);
333  channel_ref_decref(channel_ref);
334  }
335  channel_ref_set_flag(channel_ref, flag);
336  return error_ok;
337 }

340  loop_iocp_t* impl = 0;

359  }
360  channel_ref_set_data(channel_ref, per_sock);
