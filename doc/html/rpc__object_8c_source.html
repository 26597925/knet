163 int krpc_object_marshal(krpc_object_t* o, stream_t* stream, uint16_t* bytes) {
164  uint16_t size = 0;
165  int i = 0;
166  krpc_object_header_t header;
167  assert(o);
168  assert(stream);
169  memset(&header, 0, sizeof(krpc_object_header_t));
170  header.type = o->type;
171  header.length += krpc_object_get_marshal_size(o);
172  if (error_ok != stream_push(stream, &header, sizeof(krpc_object_header_t))) {
173  return error_rpc_marshal_fail;
174  }
175  if (o->type & krpc_type_string) {
176  /* 字符串 */
177  if (error_ok != stream_push(stream, o->string.str, o->string.size)) {
178  return error_rpc_marshal_fail;
179  }
180  } else if (o->type & krpc_type_vector) {
181  /* 数组 */
182  for (; i < o->vector.size; i++) {
183  /* 递归调用 */
184  if (error_ok != krpc_object_marshal(o->vector.objects[i], stream, &size)) {
185  return error_rpc_marshal_fail;
186  }
187  }
188  } else if (o->type & krpc_type_number) {
189  /* 数字 */
190  if (error_ok != stream_push(stream, &o->number, krpc_number_get_marshal_size(o))) {
191  return error_rpc_marshal_fail;
192  }
193  } else {
194  return error_rpc_marshal_fail;
195  }
196  if (bytes) {
197  *bytes = header.length;
198  }
199  return error_ok;
200 }

202 int krpc_object_unmarshal(stream_t* stream, krpc_object_t** o, uint16_t* bytes) {
203  uint16_t length = 0; /* 临时变量 */
204  int available = 0; /* 可读字节数 */
205  uint16_t consume = 0; /* 单次unmarshal消耗字节数 */
206  krpc_object_t* vo = 0; /* 数组内对象指针 */
207  krpc_object_header_t header; /* 对象协议头 */
208  assert(stream);
209  assert(o);
210  assert(bytes);
211  available = stream_available(stream);
212  if (available < sizeof(krpc_object_header_t)) {
213  /* 字节数不够 */
214  return error_rpc_not_enough_bytes;
215  }
216  if (error_ok != stream_copy(stream, &header, sizeof(header))) {
217  return error_rpc_unmarshal_fail;
218  }
219  if (header.length > (uint16_t)available) {
220  /* 字节数不够 */
221  return error_rpc_not_enough_bytes;
222  }
223  if (error_ok != stream_eat(stream, sizeof(krpc_object_header_t))) {
224  return error_rpc_unmarshal_fail;
225  }
226  /* 建立一个对象 */
227  *o = krpc_object_create();
228  assert(*o);
229  if (header.type & krpc_type_number) {
230  /* 数字 */
231  if (error_ok != stream_pop(stream, &(*o)->number, header.length - sizeof(krpc_object_header_t))) {
232  goto error_return;
233  }
234  } else if (header.type & krpc_type_string) {
235  /* 字符串 */
236  if (error_ok != krpc_string_set_size(*o, header.length - sizeof(krpc_object_header_t))) {
237  goto error_return;
238  }
239  if (error_ok != stream_pop(stream, (*o)->string.str, header.length - sizeof(krpc_object_header_t))) {
240  goto error_return;
241  }
242  (*o)->string.size = header.length - sizeof(krpc_object_header_t);
243  } else if (header.type & krpc_type_vector) {
244  /* 数组 */
245  length = header.length - sizeof(krpc_object_header_t);
246  for (; length;) {
247  /* 递归调用 */
248  if (error_ok != krpc_object_unmarshal(stream, &vo, &consume)) {
249  goto error_return;
250  }
251  if (error_ok != krpc_vector_push_back(*o, vo)) {
252  goto error_return;
253  }
254  length -= consume;
255  }
256  } else {
257  /* 未知类型 */
258  goto error_return;
259  }
260  (*o)->type = (krpc_type_e)header.type; /* 类型 */
261  *bytes = header.length; /* 消耗字节数 */
262  return error_ok;
263 error_return:
264  /* 错误处理 */
265  if (*o) {
266  krpc_object_destroy(*o);
267  *o = 0;
268  }
269  /* 告诉调用者失败，外部应关闭本连接 */
270  return error_rpc_unmarshal_fail;
271 }

506 void krpc_vector_enlarge(krpc_object_t* o) {
507  static const uint16_t DEFAULT_SIZE = 8;
508  if (!o->vector.objects) { /* 第一次建立 */
509  o->vector.max_size = DEFAULT_SIZE;
510  o->vector.objects = (krpc_object_t**)create_type(krpc_object_t,
511  sizeof(krpc_object_t*) * o->vector.max_size);
512  assert(o->vector.objects);
513  memset(o->vector.objects, 0, sizeof(krpc_object_t*) * o->vector.max_size);
514  }
515  if (o->vector.size >= o->vector.max_size) { /* 扩容 */
516  o->vector.max_size += DEFAULT_SIZE;
517  o->vector.objects = (krpc_object_t**)rcreate_type(krpc_object_t,
518  o->vector.objects, sizeof(krpc_object_t*) * o->vector.max_size);
519  assert(o->vector.objects);
520  memset(o->vector.objects + (o->vector.max_size - DEFAULT_SIZE), 0, sizeof(krpc_object_t*) * DEFAULT_SIZE);
521  }
522 }
