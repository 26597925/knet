80 int channel_ref_destroy(channel_ref_t* channel_ref) {
81  assert(channel_ref);
82  assert(channel_ref->ref_info);
83  /* 检测引用计数 */
84  if (!atomic_counter_zero(&channel_ref->ref_info->ref_count)) {
85  return error_ref_nonzero;
86  }
87  assert(channel_ref->ref_info);
88  assert(channel_ref->ref_info->loop);
89  assert(channel_ref->ref_info->channel);
90  assert(channel_ref->ref_info->stream);
91  if (channel_ref->ref_info->peer_address) {
92  address_destroy(channel_ref->ref_info->peer_address);
93  }
94  if (channel_ref->ref_info->local_address) {
95  address_destroy(channel_ref->ref_info->local_address);
96  }
97  /* 通知选取器删除管道相关资源 */
98  impl_remove_channel_ref(channel_ref->ref_info->loop, channel_ref);
99  channel_destroy(channel_ref->ref_info->channel);
100  stream_destroy(channel_ref->ref_info->stream);
101  destroy(channel_ref->ref_info);
102  destroy(channel_ref);
103  return error_ok;
104 }

132  channel_ref_set_event(channel_ref, channel_event_recv);

175 void channel_ref_close(channel_ref_t* channel_ref) {
176  loop_t* loop = 0;
177  assert(channel_ref);
178  loop = channel_ref->ref_info->loop;
179  if (loop_get_thread_id(loop) != thread_get_self_id()) {
180  /* 通知管道所属线程 */
181  log_info("close channel cross thread, notify thread[id:%d]", loop_get_thread_id(loop));
182  loop_notify_close(loop, channel_ref);
183  } else {
184  /* 本线程内关闭 */
185  log_info("close channel in loop thread[id: %d]", loop_get_thread_id(loop));
186  channel_ref_update_close_in_loop(loop, channel_ref);
187  }
188 }

208 int channel_ref_write(channel_ref_t* channel_ref, const char* data, int size) {
209  loop_t* loop = 0;
210  buffer_t* send_buffer = 0;
211  int error = error_ok;
212  assert(channel_ref);
213  assert(data);
214  assert(size);
215  if (!channel_ref_check_state(channel_ref, channel_state_active)) {
216  return error_not_connected;
217  }
218  loop = channel_ref->ref_info->loop;
219  if (loop_get_thread_id(loop) != thread_get_self_id()) {
220  /* 转到loop所在线程发送 */
221  log_info("send cross thread, notify thread[id:%d]", loop_get_thread_id(loop));
222  send_buffer = buffer_create(size);
223  buffer_put(send_buffer, data, size);
224  loop_notify_send(loop, channel_ref, send_buffer);
225  } else {
226  /* 当前线程发送 */
227  error = channel_send(channel_ref->ref_info->channel, data, size);
228  switch (error) {
229  case error_send_patial:
230  channel_ref_set_event(channel_ref, channel_event_send);
231  /* 对于调用者不是错误 */
232  error = error_ok;
233  break;
234  case error_send_fail:
235  channel_ref_close(channel_ref);
236  break;
237  default:
238  break;
239  }
240  }
241  return error;
242 }

301 channel_ref_t* channel_ref_accept_from_socket_fd(channel_ref_t* channel_ref, loop_t* loop, socket_t client_fd, int event) {
302  channel_t* acceptor_channel = channel_ref->ref_info->channel;
303  uint32_t max_send_list_len = channel_get_max_send_list_len(acceptor_channel);

306  channel_ref_t* client_ref = channel_ref_create(loop, client_channel);
307  if (event) {

312  channel_ref_set_event(client_ref, channel_event_recv);

317 void channel_ref_update_accept(channel_ref_t* channel_ref) {
318  channel_ref_t* client_ref = 0;
319  loop_t* loop = 0;
320  socket_t client_fd = 0;
321  assert(channel_ref);
322  /* 查看选取器是否有自定义实现 */
323  client_fd = impl_channel_accept(channel_ref);
324  if (!client_fd) {
325  /* 默认实现 */
326  client_fd = socket_accept(channel_get_socket_fd(channel_ref->ref_info->channel));
327  }
328  channel_ref_set_state(channel_ref, channel_state_accept);
329  channel_ref_set_event(channel_ref, channel_event_recv);
330  if (client_fd) {
331  loop = channel_ref_choose_loop(channel_ref);
332  if (loop) {
333  client_ref = channel_ref_accept_from_socket_fd(channel_ref, loop, client_fd, 0);
334  /* 设置回调 */
335  channel_ref_set_cb(client_ref, channel_ref->ref_info->cb);
336  /* 添加到其他loop */
337  loop_notify_accept(loop, client_ref);
338  } else {
339  client_ref = channel_ref_accept_from_socket_fd(channel_ref, channel_ref->ref_info->loop, client_fd, 1);
340  /* 调用回调 */
341  if (channel_ref->ref_info->cb) {
342  channel_ref->ref_info->cb(client_ref, channel_cb_event_accept);
343  }
344  }
345  }
346 }

413 void channel_ref_update(channel_ref_t* channel_ref, channel_event_e e, time_t ts) {
414  assert(channel_ref);
415  if (channel_ref_check_state(channel_ref, channel_state_close)) {
416  return;
417  }
418  if ((e & channel_event_recv) && channel_ref_check_event(channel_ref, channel_event_recv)) {
419  if (channel_ref_check_state(channel_ref, channel_state_accept)) {
420  /* 新连接 */
421  channel_ref_update_accept(channel_ref);
422  } else {
423  /* 最后一次读取到数据的时间戳（秒） */
424  channel_ref->ref_info->last_recv_ts = ts;
425  /* 读 */
426  channel_ref_update_recv(channel_ref);
427  }
428  }
429  if ((e & channel_event_send) && channel_ref_check_event(channel_ref, channel_event_send)) {
430  if (channel_ref_check_state(channel_ref, channel_state_connect)) {
431  /* 连接完成 */
432  channel_ref_update_connect(channel_ref);
433  } else {
434  /* 写 */
435  channel_ref_update_send(channel_ref);
436  }
437  }
438 }

445 loop_t* channel_ref_choose_loop(channel_ref_t* channel_ref) {
446  loop_t* loop = 0;
447  loop_t* current_loop = 0;
448  loop_balancer_t* balancer = 0;
449  assert(channel_ref);
450  current_loop = channel_ref->ref_info->loop;
451  if (!loop_get_thread_id(current_loop)) {
452  return 0;
453  }
454  balancer = loop_get_balancer(current_loop);
455  if (!balancer) {
456  return 0;
457  }
458  /* 检查是否开启了loop_balancer_out配置 */
459  if (loop_check_balance_options(channel_ref->ref_info->loop, loop_balancer_out)) {
460  loop = loop_balancer_choose(balancer);
461  if (loop == channel_ref->ref_info->loop) {
462  return 0;
463  }
464  }
465  return loop;
466 }
