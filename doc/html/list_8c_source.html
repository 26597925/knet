170 dlist_node_t* dlist_remove(dlist_t* dlist, dlist_node_t* node) {
171  assert(dlist);
172  assert(node);
173  if (!node->prev || !node->next) {
174  return 0;
175  }

177  node->next->prev = node->prev;
