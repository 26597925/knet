110 int channel_send(channel_t* channel, const char* data, int size) {
111  int bytes = 0;
112  buffer_t* send_buffer = 0;
113  assert(channel);
114  assert(data);
115  assert(size);
116  if (dlist_empty(channel->send_buffer_list)) {
117  /* 尝试直接发送 */
118  bytes = socket_send(channel->socket_fd, data, size);
119  }
120  if (bytes < 0) {
121  return error_send_fail;
122  }
123  /* 直接发送失败，或者没有发送完毕的字节放入发送链表等待下次发送 */
124  if (size > bytes) {
125  send_buffer = buffer_create(size - bytes);
126  buffer_put(send_buffer, data + bytes, size - bytes);
127  dlist_add_tail_node(channel->send_buffer_list, send_buffer);
128  /* 需要稍后发送 */
129  return error_send_patial;
130  }
131  return error_ok;
132 }

134 int channel_update_send(channel_t* channel) {
135  dlist_node_t* node = 0;
136  dlist_node_t* temp = 0;
137  buffer_t* send_buffer = 0;
138  int bytes = 0;
139  assert(channel);
140  /* 发送链表内所有数据 */
141  dlist_for_each_safe(channel->send_buffer_list, node, temp) {
142  send_buffer = (buffer_t*)dlist_node_get_data(node);
143  bytes = socket_send(channel->socket_fd, buffer_get_ptr(send_buffer), buffer_get_length(send_buffer));
144  if (bytes < 0) {
145  return error_send_fail;
146  }
147  if (buffer_get_length(send_buffer) > (uint32_t)bytes) {
148  /* 本次未发送完毕，调整buffer长度，等待下次发送 */
149  buffer_adjust(send_buffer, bytes);
150  /* 部分发送 */
151  return error_send_patial;
152  } else {
153  /* 销毁已发送节点 */
154  buffer_destroy(send_buffer);
155  dlist_delete(channel->send_buffer_list, node);
156  }
157  }
158  /* 全部发送 */
159  return error_ok;
160 }

162 int channel_update_recv(channel_t* channel) {
163  int bytes = 0;
164  int recv_bytes = 0;
165  uint32_t size = 0;
166  char* ptr = 0;
167  assert(channel);
168  if (ringbuffer_full(channel->recv_ringbuffer)) {
169  /* 读缓冲区满，关闭, 防攻击, 可根据需求调整大小 */
170  return error_recv_buffer_full;
171  }
172  for (; (size = ringbuffer_write_lock_size(channel->recv_ringbuffer));) {
173  ptr = ringbuffer_write_lock_ptr(channel->recv_ringbuffer);
174  bytes = socket_recv(channel->socket_fd, ptr, size);
175  if (bytes < 0) {
176  /* 错误，关闭 */
177  return error_recv_fail;
178  } else if (bytes == 0) {
179  /* 未接收到, 下次继续接收 */
180  ringbuffer_write_commit(channel->recv_ringbuffer, 0);
181  return error_ok;
182  } else {
183  recv_bytes += bytes;
184  /* 接收到 */
185  ringbuffer_write_commit(channel->recv_ringbuffer, (uint32_t)bytes);
186  }
187  }
188  if (!recv_bytes) {
189  /* 本次不能完成接收，非关闭类错误 */
190  return error_recv_nothing;
191  }
192  return error_ok;
193 }
