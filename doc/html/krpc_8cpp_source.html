31 krpc_gen_t::krpc_gen_t(int argc, char** argv) {
32  // 简单的解析命令行
33  for (int pos = 1; pos < argc; pos++) {
34  if (get_option(pos, argc, argv, "lang", "l") ||
35  get_option(pos, argc, argv, "file", "f") ||
36  get_option(pos, argc, argv, "dir", "d") ||
37  get_option(pos, argc, argv, "name", "n")) {
38  pos++;
39  }
40  }
41  // 开始解析协议文件
42  _parser = new krpc_parser_t(0, _options["dir"].c_str(),

47 bool krpc_gen_t::get_option(int pos, int argc, char** argv, const char* detail,
48  const char* brief) {
49  std::string detail_option = "-";
50  std::string brief_option = "-";
51  detail_option += detail;
52  brief_option += brief;
53  if ((argv[pos] == detail_option) || (argv[pos] == brief_option)) {
54  if (pos + 1 < argc) {
55  _options[detail] = argv[pos + 1];
56  return true;
57  } else {
58  std::cout << "need specify '" << detail << "'" << std::endl;
59  exit(0);
60  }
61  }
62  return false;
63 }
