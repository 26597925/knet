63 int loop_balancer_attach(loop_balancer_t* balancer, loop_t* loop) {
64  dlist_node_t* node = 0;
65  dlist_node_t* temp = 0;
66  loop_info_t* loop_info = 0;
67  int error = error_ok;
68  assert(balancer);
69  assert(loop);
70  lock_lock(balancer->lock);
71  dlist_for_each_safe(balancer->loop_info_list, node, temp) {
72  loop_info = (loop_info_t*)dlist_node_get_data(node);
73  if (loop_info->loop == loop) {
74  error = error_loop_attached;
75  goto unlock_return;
76  }
77  }
78  loop_info = create(loop_info_t);
79  assert(loop_info);
80  memset(loop_info, 0, sizeof(loop_info_t));
81  loop_info->loop = loop;
82  dlist_add_tail_node(balancer->loop_info_list, loop_info);
83  loop_set_balancer(loop, balancer);
84 unlock_return:
85  lock_unlock(balancer->lock);
86  return error;
87 }

92  dlist_node_t* temp = 0;

95  assert(balancer);

97  lock_lock(balancer->lock);
98  dlist_for_each_safe(balancer->loop_info_list, node, temp) {
99  loop_info = (loop_info_t*)dlist_node_get_data(node);
100  if (loop_info->loop == loop) {
101  found = node;
102  break;

105  if (found) {
106  loop_set_balancer(loop_info->loop, 0);
107  destroy(loop_info);
108  dlist_delete(balancer->loop_info_list, found);
109  } else {
110  error = error_loop_not_found;
111  }
112  lock_unlock(balancer->lock);
113  return error;

116 loop_t* loop_balancer_choose(loop_balancer_t* balancer) {
117  dlist_node_t* node = 0;
118  dlist_node_t* temp = 0;
119  dlist_t* channel_list = 0;
120  loop_info_t* found = 0;
121  loop_info_t* loop_info = 0;
122  int channel_count = INT_MAX;
123  int count = 0;
124  assert(balancer);
125  lock_lock(balancer->lock);
126  /* 选取当前活跃管道数最少的loop_t */
127  dlist_for_each_safe(balancer->loop_info_list, node, temp) {
128  loop_info = (loop_info_t*)dlist_node_get_data(node);
129  /* 是否开启loop_balancer_in配置 */
130  if (loop_check_balance_options(loop_info->loop, loop_balancer_in)) {
131  channel_list = loop_get_active_list(loop_info->loop);
132  count = dlist_get_count(channel_list);
133  if (count < channel_count) {
134  found = loop_info;
135  channel_count = count;
136  }
137  }
138  }
139  if (found) {
140  /* 记录被选取次数，保留 */
141  found->choose++;
142  }
143  lock_unlock(balancer->lock);
144  if (found) {
145  return found->loop;
146  }
147  return 0;
148 }
