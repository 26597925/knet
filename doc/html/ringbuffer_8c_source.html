65 int ringbuffer_eat(ringbuffer_t* rb, uint32_t size) {
66  assert(rb);
67  if (rb->lock_size || rb->lock_type) {
68  return error_recvbuffer_locked;
69  }
70  if (size > rb->count) {
71  return error_recvbuffer_not_enough;
72  }
73  rb->count -= size;
74  rb->read_pos = (rb->read_pos + size) % rb->max_size;
75  return error_ok;
76 }

78 uint32_t ringbuffer_read(ringbuffer_t* rb, char* buffer, uint32_t size) {
79  uint32_t i = 0;
80  assert(rb);
81  assert(buffer);
82  assert(size);
83  size = min(rb->count, size);
84  for (; i < size; i++) {
85  buffer[i] = rb->ptr[rb->read_pos];
86  rb->read_pos = (rb->read_pos + 1) % rb->max_size;
87  }
88  rb->count -= size;
89  return size;
90 }

118 uint32_t ringbuffer_read_lock_size(ringbuffer_t* rb) {
119  assert(rb);

121  return 0;
