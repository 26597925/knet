79 void ktimer_loop_destroy(ktimer_loop_t* ktimer_loop) {
80  int i = 0;
81  ktimer_t* timer = 0;
82  dlist_node_t* node = 0;
83  dlist_node_t* temp = 0;
84  assert(ktimer_loop);

138 int ktimer_loop_run_once(ktimer_loop_t* ktimer_loop) {
139  dlist_node_t* node = 0;
140  dlist_node_t* temp = 0;
141  dlist_t* timers = 0;
142  ktimer_t* timer = 0;
143  time_t ms = time_get_milliseconds(); /* 当前时间戳（毫秒） */
144  int count = 0;
145  assert(ktimer_loop);
146  timers = ktimer_loop->ktimer_wheels[ktimer_loop->slot];
147  dlist_for_each_safe(timers, node, temp) {
148  timer = (ktimer_t*)dlist_node_get_data(node);
149  if (!_ktimer_check_stop(timer)) {
150  /* 处理定时器 */
151  if (ktimer_check_timeout(timer, ms)) {
152  count++;
153  }
154  } else {
155  /* 销毁 */
156  ktimer_destroy(timer);
157  }
158  }
159  /* 下一个槽位 */
160  ktimer_loop->slot = (ktimer_loop->slot + 1) % ktimer_loop->max_slot;
161  /* 记录上次tick时间戳 */
162  ktimer_loop->last_tick = ms;
163  return count;
164 }

228 int ktimer_check_timeout(ktimer_t* timer, time_t ms) {
229  dlist_node_t* node = 0;
230  time_t tick_intval = 0;
231  ktimer_loop_t* ktimer_loop = 0;
232  assert(timer);
233  ktimer_loop = timer->ktimer_loop;
234  assert(ktimer_loop);
235  tick_intval = ktimer_loop_get_tick_intval(ktimer_loop);
236  if (timer->ms > ms) {
237  /* 在将来的时间到期，检测是否可以触发 */
238  if (timer->ms - ms >= tick_intval) {
239  /* 距离下次超时时间大于（等于）一个tick间隔, 可以被下次触发, 调整到后续槽位槽位 */
240  node = _ktimer_loop_remove_timer(timer);
241  _ktimer_loop_add_ktimer_node(timer->ktimer_loop, node, timer->ms - ms);
242  return 0;
243  } else {
244  /* 距离下次超时时间小于一个tick间隔, 如果下一轮触发已经多超时了一个tick间隔 */
245  }
246  }
247  if (timer->type == ktimer_type_times) {
248  /* 先改变次数 */
249  timer->ms = ms + timer->intval;
250  timer->current_times++;
251  }
252  timer->cb(timer, timer->data);
253  if (_ktimer_check_stop(timer)) {
254  /* 回调内调用ktimer_stop() */
255  ktimer_destroy(timer);
256  return 1;
257  }
258  if (timer->type == ktimer_type_once) {
259  ktimer_destroy(timer);
260  } else if (timer->type == ktimer_type_period) {
261  timer->ms = ms + timer->intval;
262  node = _ktimer_loop_remove_timer(timer);
263  _ktimer_loop_add_ktimer_node(timer->ktimer_loop, node, timer->intval);
264  } else if (timer->type == ktimer_type_times) {
265  if (timer->times <= timer->current_times) {
266  ktimer_destroy(timer);
267  } else {
268  node = _ktimer_loop_remove_timer(timer);
269  _ktimer_loop_add_ktimer_node(timer->ktimer_loop, node, timer->intval);
270  }
271  }
272  return 1;
273 }
